Run Locally

# Clone & venv
git clone https://github.com/tmutoni/persona-aware-AI-agent-system.git  
cd persona-aware-AI-agent-system  
python -m venv .venv && source .venv/bin/activate

# Install & run
pip install -r agent_core/requirements.txt -r web_demo/requirements.txt  
python agent_core/server.py    # tab 1  
streamlit run web_demo/app.py  # tab 2
